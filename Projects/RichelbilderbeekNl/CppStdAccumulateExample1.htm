(C++) std::accumulate example 1: summing a std::vector of integers

main.cpp

 

#include <cassert>
#include <vector>
#include <numeric>

int main()
{
  //Create a std::vector
  std::vector<int> v;
  for (int i=0; i!=10; ++i) { v.push_back(i); }

  //Sum the std::vector
  const int sum = std::accumulate(v.begin(), v.end(), 0 ); // '0' is the initial value

  //Assume std::accumulate works correctly
  assert(sum == 45);
}
