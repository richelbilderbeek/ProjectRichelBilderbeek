#include <iostream>
#include "apfloat.h"

using namespace std;

int main()
{
  double x = 1.0;
  apfloat y = 1.0;
  for (int i=0; i!=20; ++i)
  {
    x /= 1.0e50;
    y /= 1.0e50;
    std::cout << x << " " << y << '\n';
  }
}
