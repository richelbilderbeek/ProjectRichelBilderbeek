(C++) std::bitset

 

std::bitset is a container for bits.

 

#include <bitset>
#include <iostream>

int main()
{
  const int sz = 8;
  std::bitset<sz> x;
  for (int i=0; i!=sz; ++i)
  {
    std::cout << x << '\t' << x.to_ulong() << '\n';
    x[i] = true;
  }
  std::cout << x << '\t' << x.to_ulong() << '\n';
}
