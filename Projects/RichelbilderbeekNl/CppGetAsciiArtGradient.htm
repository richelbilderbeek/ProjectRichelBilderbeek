(C++) GetAsciiArtGradient

 

ASCII art code snippet to get an ASCII art gradient.

 

#include <vector>

//From http://www.richelbilderbeek.nl/CppGetAsciiArtGradient.htm
const std::vector<char> GetAsciiArtGradient()
{
  std::vector<char> chars;
  chars.push_back('M');
  chars.push_back('N');
  chars.push_back('m');
  chars.push_back('d');
  chars.push_back('h');
  chars.push_back('y');
  chars.push_back('s');
  chars.push_back('o');
  chars.push_back('+');
  chars.push_back('/');
  chars.push_back(':');
  chars.push_back('-');
  chars.push_back('.');
  chars.push_back('`');
  chars.push_back(' ');
  return chars;
}
