#include <iostream>

//Example adapted from https://en.wikipedia.org/wiki/Gcov
int main()
{
  for (int i = 1; i!=10; ++i)
  {
    if (i % 3 == 0)
    {
      std::cout << i << " is divisible by 3\n";
    }
    if (i % 11 == 0)
    {
      std::cout << i << " is divisible by 11\n";
    }
  }
}

/* Screen output:

3 is divisible by 3
6 is divisible by 3
9 is divisible by 3
Press <RETURN> to close this window...

*/
