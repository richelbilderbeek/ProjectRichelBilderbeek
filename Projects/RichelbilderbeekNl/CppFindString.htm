(C++) FindString

 

Find a std::string in a std::string.

 

#include <cassert>
#include <string>

int main ()
{
  const std::string searchme = "BILDERBIKKEL WAS HERE";
  assert(searchme.find("BILDERBIKKEL") != std::string::npos);
  assert(searchme.find("RICHEL") == std::string::npos);
}
