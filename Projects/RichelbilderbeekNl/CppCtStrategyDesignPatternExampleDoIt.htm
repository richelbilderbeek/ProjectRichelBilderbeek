(C++) Compile-time Strategy Design Pattern example: DoIt

 

Compile-time Strategy Design Pattern example: DoIt is a compile-time Strategy Design Pattern.

main.cpp

 

#include <iostream>

///A compile-time Strategy Design Pattern
enum Policy { A, B };

template <Policy> struct Strategy
{
  static void DoIt();
};

template<> void Strategy<A>::DoIt()
{
  std::cout << "Do it the A way\n";
}

template<> void Strategy<B>::DoIt()
{
  std::cout << "Do it the B way\n";
}

int main()
{
  const Strategy<A> x; x.DoIt();
  const Strategy<B> y; y.DoIt();
}
