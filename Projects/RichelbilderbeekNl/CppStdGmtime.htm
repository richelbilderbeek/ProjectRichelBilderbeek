(C++) std::gmtime

 

std::gmtime is a time function.

 

#include <ctime>
#include <iostream>

int main ()
{
  std::time_t t;
  std::time(&t);
  const std::tm * const u = std::gmtime(&t);
  std::cout
    << u->tm_hour
    << ":"
    << u->tm_min;
}
