qtdialog.h

 

#ifndef QTDIALOG_H
#define QTDIALOG_H

#include <QDialog>

namespace Ui {
  class QtDialog;
}

struct QStandardItemModel;

class QtDialog : public QDialog
{
  Q_OBJECT
  
public:
  explicit QtDialog(QWidget *parent = 0);
  ~QtDialog();
  
private slots:
  void on_button_clicked();

private:
  Ui::QtDialog *ui;
  QStandardItemModel * const m_model;
};

#endif // QTDIALOG_H

 

 

 

 

 

qtdialog.cpp

 

#include "qtdialog.h"

#include <QStandardItemModel>
#include <QStyledItemDelegate>
#include "ui_qtdialog.h"

QtDialog::QtDialog(QWidget *parent) :
  QDialog(parent),
  ui(new Ui::QtDialog),
  m_model(new QStandardItemModel)
{
  ui->setupUi(this);
  ui->list_view->setModel(m_model);
  ui->list_view->setItemDelegate(new QStyledItemDelegate);
}

QtDialog::~QtDialog()
{
  delete ui;
}

void QtDialog::on_button_clicked()
{
  QStandardItem * const item = new QStandardItem;
  item->setText("");
  m_model->appendRow(item);
  ui->list_view->scrollToBottom();
  ui->list_view->setCurrentIndex(item->index());
  ui->list_view->edit(item->index());

}

 

 

 

 

 

qtmain.cpp

 

#include <QApplication>
#include "qtdialog.h"

int main(int argc, char *argv[])
{
  QApplication a(argc, argv);
  QtDialog w;
  w.show();
  return a.exec();
}
