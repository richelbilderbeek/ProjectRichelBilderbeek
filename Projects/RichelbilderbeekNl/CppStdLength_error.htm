(C++) std::length_error

 

std::length_error (a derived class from std::runtime_error) is thrown when a too big std::string is created.

 

#include <iostream>
#include <stdexcept>
#include <string>

int main()
{
  std::string s;
  try
  {
    s.resize(s.max_size()+1);
  }
  catch (std::length_error& e)
  {
    std::cout << e.what();
  }
}
