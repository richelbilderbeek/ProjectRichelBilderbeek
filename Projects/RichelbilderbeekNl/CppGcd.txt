int GreatestCommonDivisor(int x, int y)
{
  while (y != 0)
  {
    const int temp = y;
    y = x % y;
    x = temp;
  }
  return x;
}
//---------------------------------------------------------------------------


//---------------------------------------------------------------------------
// Using recursion
//---------------------------------------------------------------------------
int GreatestCommonDivisor(const int x, const int y)
{
  if (y==0) return x;
  return GreatestCommonDivisor(y,x%y);
}
