#include <cassert>
#include <map>
#include <vector>

//Modeled after the PHP function with the same name
template <class T,class U>
const std::vector<T> array_keys(const std::map<T,U>& m)
{
  std::vector<T> v;
  for (const std::pair<T,U>& p: m) { v.push_back(p.first); }
  return v;
}

//Modeled after the PHP function with the same name
