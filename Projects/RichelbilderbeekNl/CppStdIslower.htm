(C++) std::islower

 

std::islower is an STL function to check if a character is a lowercase alphabetic letter.

 

  • Download the Qt Creator project 'CppIslower' (zip)

 

#include <cassert>
#include <cctype>

int main()
{
  assert(std::islower('a'));
  assert(!std::islower('A'));
  assert(!std::islower('1'));
  assert(!std::islower('@'));
}
