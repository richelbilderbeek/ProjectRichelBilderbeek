// derivatv.cpp

#include <iostream>
#include "symbolicc++.h"
using namespace std;

int main(void)
{
   int i;
   Symbolic x("x"), y, z("z");

   y = 1/(1-x) + 2*(x^3) - z;
   cout << "y = " << y << endl;

   for(i=0;i<8;i++)
   { y = df(y,x); cout << "y = " << y << endl; }
   cout << endl;

   Symbolic u, v("v");

   u = (v^(Symbolic(3)/5))/3 - 2*(v^(Symbolic(1)/5))/7 + Symbolic(1)/6;
   cout << "u = " << u << endl;

   for(i=0;i<8;i++)
   { u = df(u,v); cout << "u = " << u << endl; }
   return 0;
}
