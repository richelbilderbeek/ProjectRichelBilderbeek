(C++) DotMatrixVcl

 

DotMatrix (VCL) is a tool to display text as dot-matrix characters (5 (width) x 7 (height) pixels, without spacer pixels), like on your microwave display.

 

DotMatrix (VCL) is the predecessor of DotMatrix.

void DotMatrix(
  Extctrls::TImage * const image,
  const int x_co,
  const int y_co,
  const char c)
{
  const boost::array<boost::array<int,5> ,7> v(GetMatrix(c));

  for (int y=0; y!=7; ++y)
  {
    for (int x=0; x!=5; ++x)
    {
      const int x1 = x_co + x;
      const int y1 = y_co + y;
      if (v[y][x] == 1)
        SetPixel(image,x1,y1,  0,  0,  0);
      else
        SetPixel(image,x1,y1,255,255,255);
    }
  }

}
//---------------------------------------------------------------------------
void DotMatrix(
  Extctrls::TImage * const image,
  const int x_co,
  const int y_co,
  const std::string& s)
{
  const int sz = s.size();
  for (int x=0; x!=sz; ++x)
  {
    DotMatrix(image,x_co + (x*6),y_co,s[x]);
  }

}
//---------------------------------------------------------------------------
void DotMatrix(
  Extctrls::TImage * const image,
  const int x_co,
  const int y_co,
  const std::vector<std::string>& v)
{
  const int n_rows = v.size();
  for (int y=0; y!=n_rows; ++y)
  {
    DotMatrix(image,x_co,y_co+(y*8),v[y]);
  }

}

//From http://www.richelbilderbeek.nl
void SetPixel(
  Extctrls::TImage * const image,
  const int x,
  const int y,
  const unsigned char red,
  const unsigned char green,
  const unsigned char blue)
{
  assert(image!=0 && "Image is NULL");
  assert(image->Picture->Bitmap!=0 && "Bitmap is NULL");
  assert(image->Picture->Bitmap->PixelFormat == pf24bit && "Bitmap must be 24 bit");
  assert( x >= 0 && "x coordinat is below zero");
  assert( y >= 0 && "y coordinat is below zero");
  assert( x < image->Picture->Bitmap->Width  && "x coordinat is beyond image width");
  assert( y < image->Picture->Bitmap->Height && "y coordinat is beyond image height");

  unsigned char * const line
    = static_cast<unsigned char *>(image->Picture->Bitmap->ScanLine[y]);

  line[x*3+2] = red;
  line[x*3+1] = green;
  line[x*3+0] = blue;
}

  assert( x2 <= image->Picture->Bitmap->Width  && "x2 coordinat is beyond image width");
  assert( y < image->Picture->Bitmap->Height && "y coordinat is beyond image height");

    myLine[x*3+2] = red;

    myLine[x*3+0] = blue;

{

  TStrings* const s = const_cast<TStrings*>(strings);

  const int sz = s->Count;
  std::vector<std::string> v(sz);

  for (int i=0; i!=sz; ++i)
  {
    v[i]=s->operator[](i).c_str();
  }
  return v;
}

const int FindLongestStringLength(const std::vector<std::string>& v)
{
  if (v.empty()) return 0;
  int length = v[0].size();

  const int sz = v.size();
  for (int i=1; i!=sz; ++i) //1 as 0 has already been read
  {
    const int this_length = v[i].size();
    if (this_length > length) length = this_length;
  }
  return length;
}
