(C++) std::setbase

 

std::setbase is a stream manipulator to change the output of numbers to the octal, decimal or hexadecimal number system.

 

  • Download the Qt Creator project 'CppSetbase' (zip)

 

 

 

 

 

#include <iostream>
#include <iomanip>
int main()
{
  for (int i=0; i!=20; ++i)
  {
    std::cout
      << std::setbase(10) << i << "(dec) "
      << std::setbase(8)  << i << "(oct) "
      << std::setbase(16) << i << "(hex)"
      << '\n';
  }
}
