Question #8: GetSum

 

Replace the for-loop. You will need:

  • std::accumulate

 

#include <vector>

const int GetSum (const std::vector<int>& v)
{
  const int sz = static_cast<int>(v.size());
  const int sum = 0;
  for (int i=0; i!=sz; ++i)
  {
    sum+=v[i];
  }
  return sum;
}

 

 

 

 

 

Answer

 

#include <algorithm>
#include <numeric>
#include <vector>

const int GetSum (const std::vector<int>& v)
{
  return std::accumulate(v.begin(),v.end(),0);
}
