(C++) std::exp

 

std::exp is am STL math function to obtain the natural logarithm of a value.

 

#include <cmath>
#include <iostream>

int main()
{
  //Set output to fixed length
  std::cout << std::fixed;
  for (double x = 0.0; x < 1.0; x+=0.1)
  {
    const double y = std::exp(x);
    std::cout
      << "std::exp(" << x << ") = "
      << y << '\n';
  }
}
